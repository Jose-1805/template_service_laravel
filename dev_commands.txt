alias octane_status="docker exec -it $container php artisan octane:status"
alias db_seed="docker exec -it $container php artisan db:seed"
alias connect="docker exec -it $container /bin/sh"

alias dev_dir_permissions="docker exec -it $container sh ./docker/commands/dev_dir_permissions.sh"
